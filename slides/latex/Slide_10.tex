\begin{frame}[t]

	\frametitle{Raising a number to an Imaginary power}

	% If we let \(n = 1\), then we have;\\~
	% \begin{equation} \label{eqn_slide_10_a}
	% e^{j}
	% \end{equation}

	% The result of this is that \(e\) gets multiplied by \(j\) once. That is;
	% \begin{equation} \label{eqn_slide_10_b}
	% e^{j} = j \times e
	% \end{equation}

	So, for \(n = 0\) through \(n = 3\) we get;
	\begin{flalign} \label{eqn_slide_10_c}
	&e^{j0} = 1 &= 1\\
	&e^{j1} = j \times e &= je\\
	&e^{j2} = j \times j \times e = j^{2}e &= -e \\
	&e^{j3} = j \times j \times j \times e = j^{3}e &= -je \\
	&e^{j4} = j \times j \times j \times j \times e = j^{4}e &= je
	\end{flalign}

\end{frame}
